05 The new pattern — chained fetching

Fetch one thing → user clicks → fetch another thing

Two useEffects, working together
First useEffect — fires once on mount. Loads the list.
Second useEffect — fires when the user clicks. Loads the details.

This is the master/detail pattern. You'll use it a lot in real apps.

15 App 1 — final code (reference)
posts-viewer.tsx
import { useEffect, useState } from "react"; type Post = Readonly<{ id: number; title: string; body: string }>; type Comment = Readonly<{ id: number; postId: number; name: string; email: string; body: string }>; export function PostsViewer() { const [posts, setPosts] = useState<Post[]>([]); const [selectedPost, setSelectedPost] = useState<Post | null>(null); const [comments, setComments] = useState<Comment[]>([]); useEffect(() => { fetch("https://jsonplaceholder.typicode.com/posts") .then(r => r.json()).then(setPosts); }, []); useEffect(() => { if (!selectedPost) return; fetch(`https://jsonplaceholder.typicode.com/posts/${selectedPost.id}/comments`) .then(r => r.json()).then(setComments); }, [selectedPost?.id]); return ( <div className="flex gap-4 p-4"> <ul className="w-1/3 space-y-2"> {posts.map(p => ( <li key={p.id} onClick={() => setSelectedPost(p)} className="cursor-pointer rounded-md border border-gray-200 bg-white p-3 hover:bg-gray-50"> {p.title} </li> ))} </ul> <div className="flex-1"> {selectedPost ? ( <div className="space-y-4"> <h2 className="text-xl font-bold">{selectedPost.title}</h2> <p>{selectedPost.body}</p> <h3 className="font-bold">Comments</h3> <ul className="space-y-2"> {comments.map(c => ( <li key={c.id} className="rounded-md bg-gray-50 p-3"> <strong>{c.name}</strong> — {c.email} <p>{c.body}</p> </li> ))} </ul> </div> ) : ( <p className="text-gray-500">Click a post to see details</p> )} </div> </div> ); }

26 App 2 — final code (reference)
pokemon-explorer.tsx
import { useEffect, useState } from "react"; type PokemonListItem = Readonly<{ name: string; url: string }>; type PokemonDetail = Readonly<{ name: string; height: number; weight: number; sprites: { front_default: string }; types: { type: { name: string } }[]; stats: { base_stat: number; stat: { name: string } }[]; }>; export function PokemonExplorer() { const [list, setList] = useState<PokemonListItem[]>([]); const [selectedName, setSelectedName] = useState<string | null>(null); const [detail, setDetail] = useState<PokemonDetail | null>(null); useEffect(() => { fetch("https://pokeapi.co/api/v2/pokemon?limit=20") .then(r => r.json()).then(data => setList(data.results)); }, []); useEffect(() => { if (!selectedName) return; fetch(`https://pokeapi.co/api/v2/pokemon/${selectedName}`) .then(r => r.json()).then(setDetail); }, [selectedName]); return ( <div className="flex gap-4 p-4"> <ul className="w-1/3 space-y-2"> {list.map(p => ( <li key={p.name}> <button onClick={() => setSelectedName(p.name)} className="w-full rounded-md border border-gray-200 bg-white p-3 capitalize hover:bg-gray-50"> {p.name} </button> </li> ))} </ul> <div className="flex-1"> {detail ? ( <div className="rounded-xl border border-gray-200 bg-white p-6 text-center"> <img src={detail.sprites.front_default} alt={detail.name} className="mx-auto h-40" /> <h2 className="text-2xl font-bold capitalize">{detail.name}</h2> <div className="mt-2 flex justify-center gap-2"> {detail.types.map(t => ( <span key={t.type.name} className="rounded-full bg-blue-100 px-3 py-1 text-sm capitalize"> {t.type.name} </span> ))} </div> <p className="mt-2 text-sm text-gray-500"> Height: {detail.height} · Weight: {detail.weight} </p> <ul className="mt-4 space-y-1 text-left max-w-xs mx-auto"> {detail.stats.map(s => ( <li key={s.stat.name} className="capitalize"> {s.stat.name}: {s.base_stat} </li> ))} </ul> </div> ) : ( <p className="text-gray-500">Click a Pokémon to see details</p> )} </div> </div> ); }
